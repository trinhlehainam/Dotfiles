- Install neovim : 	     sudo apt instlal neovim
- Make nvim directory :      mkdir ~/.config/nvim
- Make nvim init file :      touch ~/.config/nvim/init.vim
- Copy plugin manager file : vim-plugs 
- Make plugsin file :	     touch ~/.config/nvim/vim-plug/plugins.vim

- Copy this command to plugsin.vim:

" auto-install vim-plug
if empty(glob('~/.config/nvim/autoload/plug.vim'))
  silent !curl -fLo ~/.config/nvim/autoload/plug.vim --create-dirs
    \ https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim
  "autocmd VimEnter * PlugInstall
  "autocmd VimEnter * PlugInstall | source $MYVIMRC
endif

call plug#begin('~/.config/nvim/autoload/plugged')

    " Better Syntax Support
    Plug 'sheerun/vim-polyglot'
    " File Explorer
    Plug 'scrooloose/NERDTree'
    " Auto pairs for '(' '[' '{'
    Plug 'jiangmiao/auto-pairs'

call plug#end()

- Source | include file to init.vim:
source $HOME/.config/nvim/vim-plug/plugins.vim

- Open nvim and install plugins.

*** if ERROR : [vim-plug] `git` executable not found   -->  install git
